custom_button = {
	name = th_open_danmaku_battle_button
	tooltip = th_open_danmaku_battle_button_desc
	potential = { th_has_danmaku_duels_unlocked = yes }
	trigger = {  }
	effect = {
		if = {
			limit = { th_is_in_a_danmaku_duel = yes }
			hidden_effect = {
				if = {
					limit = { has_country_flag = th_opened_danmaku_battle_window }
					clr_country_flag = th_opened_danmaku_battle_window
				}
				else = { set_country_flag = th_opened_danmaku_battle_window }
			}
		}
		else = {
			if = {
				limit = {
					custom_trigger_tooltip = {
						tooltip = th_is_in_a_danmaku_duel_tt
						NOT = { has_ruler_flag = th_ruler_is_in_a_danmaku_duel }
					}
				}
				custom_tooltip = th_open_danmaku_battle_button_at
			}
			if = {
				limit = { is_lesser_in_union = yes }
				custom_tooltip = th_open_danmaku_battle_button_bt
			}
		}
	}
}
