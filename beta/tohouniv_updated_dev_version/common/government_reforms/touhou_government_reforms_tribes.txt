	}
	effect = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					NOT = { nationalism = 1 }
					is_core = ROOT
					th_hellfairy_has_no_stack = yes
				}
				hellfairy_inititiate_berserk_stack = yes
			}
		}
	}
	post_removed_effect = {
		remove_country_modifier = th_hellfairy_provinces_to_overload
		every_owned_province = {
			limit = {
				has_province_modifier = th_province_overload
			}
			remove_province_modifier = th_province_overload
		}
		remove_country_modifier = th_hellfairy_army_berzerk
	}
}

TH_fairies_become_absolute_monarchy_reform = {
	icon = "queen"
	allow_normal_conversion = yes
	potential = {
		primary_culture = Fairy
	}
	effect = {
		set_country_flag = populists_in_government
		lose_reforms = 2
		change_government = monarchy
		if = {
			limit = {
				th_is_or_was_tag = {
					tag = CIN
				}
			}
			add_government_reform = Gensokyan_cirno_kingdom_government
		}
		else_if = {
			limit = {
				th_is_or_was_tag = {
					tag = DAY
				}
			}
			add_government_reform = Gensokyan_dai_fairy_geniocracy
		}
		else = {
			add_government_reform = Gensokyan_monarchy_fairy_government
		}
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_global_flag = th_royal_fairy_challenge_happened_once }
				}
				set_global_flag = th_royal_fairy_challenge_is_about_to_happen
				if = {
					limit = {
						th_is_or_was_tag = {
							tag = CIN
						}
					}
					country_event = {
						id = flavor_cin.3
						days = 365
					}
				}
				else = {
					country_event = {
						id = rfc_events.15
						days = 365
					}
				}
			}
		}
	}
	ai = {
		factor = 500
	}
}
