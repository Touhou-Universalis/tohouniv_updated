#Challenge to Danmaku Duel
th_challenge_to_danmaku_duel = {
	category = th_diplo_action
	alert_index = 5
	alert_tooltip = th_challenge_to_danmaku_duel_alert_tooltip
	require_acceptance = yes
	is_visible = { th_has_danmaku_duels_unlocked = yes }
	is_allowed = {
		if = {
			limit = { has_ruler_modifier = th_danmaku_incapacitated }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_ROOT_RULER_IS_INCAPACITATED_TT
				NOT = { has_ruler_modifier = th_danmaku_incapacitated }
			}
		}
		if = {
			limit = { FROM = { has_ruler_modifier = th_danmaku_incapacitated } }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_FROM_RULER_IS_INCAPACITATED_TT
				FROM = { NOT = { has_ruler_modifier = th_danmaku_incapacitated } }
			}
		}		
		if = {
			limit = { NOT = { has_global_flag = th_initialized_danmaku_database } }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_DUEL_DUELS_HAVE_NOT_BEEN_INITIATED_YET_TT
				has_global_flag = th_initialized_danmaku_database
			}
		}
		if = {
			limit = { NOT = { has_ruler_flag = th_danmaku_ruler_is_set } }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_ROOT_RULER_NOT_PREPARED_YET_TT
				has_ruler_flag = th_danmaku_ruler_is_set
			}
		}
		if = {
			limit = { FROM = { NOT = { has_ruler_flag = th_danmaku_ruler_is_set } } }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_FROM_RULER_NOT_PREPARED_YET_TT
				FROM = { has_ruler_flag = th_danmaku_ruler_is_set }
			}
		}
		if = {
			limit = { th_is_in_a_danmaku_duel = yes }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_DUEL_ROOT_IS_ALREADY_IN_A_DUEL_TT
				th_is_not_in_a_danmaku_duel = yes
			}
		}
		if = {
			limit = { FROM = { th_is_in_a_danmaku_duel = yes } }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_DUEL_FROM_IS_ALREADY_IN_A_DUEL_TT
				th_is_not_in_a_danmaku_duel = yes
			}
		}
		if = {
			limit = { FROM = { th_has_not_danmaku_duels_unlocked = yes } }
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_DUEL_FROM_IS_NOT_VALID_FOR_DUEL_TT
				th_has_danmaku_duels_unlocked = yes
			}
		}
		#if = {
		#	limit = {
		#		alliance_with = FROM
		#	}
		#	custom_trigger_tooltip = {
		#		tooltip = TH_FROM_IS_OUR_ALLY
		#		NOT = { alliance_with = FROM }
		#	}
		#}
		if = {
			limit = {
				war_with = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_AT_WAR_WITH_FROM
				NOT = { war_with = ROOT }
			}
		}
		if = {
			limit = {
				has_regency = yes
			}
			custom_trigger_tooltip = {
				tooltip = TH_HAS_REGENCY
				has_regency = no
			}
		}
		if = {
			limit = {
				FROM = { has_regency = yes }
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_HAS_REGENCY
				FROM = { has_regency = no }
			}
		}
	}
	on_accept = {
		th_create_danmaku_duel_against_country = {
			country_1 = ROOT
			country_2 = FROM
		}
	}
	ai_acceptance = {
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_hostile
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = 50
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_outraged
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = 50
			}
		}
		add_entry = {
			name = DIPLO_IS_RIVAL
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_rivalry
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			multiply_variable = {
				which = ai_value
				value = 100
			}
		}
		add_entry = {
			name = DIPLO_IS_THREAT
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_threatened
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = 25
			}
		}
		add_entry = {
			name = DIPLO_HISTORICAL_RIVAL
			limit = {
				OR = { 
					historical_rival_with = FROM
					FROM = { historical_rival_with = ROOT }
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = 1000
			}
		}
		add_entry = {
			name = TH_BASE_OBEDIENCE
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}			
			multiply_variable = {
				which = ai_value
				value = 25
			}
		}
		add_entry = {
			name = TH_HAS_ALLIANCE
			limit = {
				FROM = {
					alliance_with = ROOT
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			multiply_variable = {
				which = ai_value
				value = -50
			}
		}
	}
	ai_will_do = {
		always = no
	}
}

th_challenge_to_danmaku_duel_old = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		th_has_touhou_technology = yes
		is_free_or_tributary_trigger = yes
		FROM = {
			OR = {
				is_free_or_tributary_trigger = yes
				AND = {
					is_subject_of = ROOT
					is_subject_of_type = th_rebellious_subject
				}
			}
			th_has_touhou_technology = yes
			has_regency = no
		}
	}

	is_allowed = {
		if = {
			limit = {
				NOT = { mil_power = 25 }
			}
			custom_trigger_tooltip = {
				tooltip = TH_ROOT_REQUIRES_25_MIL_TT
				mil_power = 25
			}
		}
		if = {
			limit = {
				NOT = { has_government_attribute = th_decreased_danmaku_duel_refusal }
				FROM = {
					has_country_flag = th_was_already_danmaku_challenged_by_@ROOT
					NOT = {
						had_country_flag = {
							flag = th_was_already_danmaku_challenged_by_@ROOT
							days = 7300
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_HAS_BEEN_CHALLENGED_IN_THE_LAST_20_YEARS_TT
				FROM = {				
					OR = {
						has_government_attribute = th_decreased_danmaku_duel_refusal
						NOT = { has_country_flag = th_was_already_danmaku_challenged_by_@ROOT }
						had_country_flag = {
							flag = th_was_already_danmaku_challenged_by_@ROOT
							days = 7300
						}
					}
				}
			}
		}
		if = {
			limit = {
				has_government_attribute = th_decreased_danmaku_duel_refusal
				FROM = {
					has_country_flag = th_was_already_danmaku_challenged_by_@ROOT
					NOT = {
						had_country_flag = {
							flag = th_was_already_danmaku_challenged_by_@ROOT
							days = 3650
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_HAS_BEEN_CHALLENGED_IN_THE_LAST_10_YEARS_TT
				FROM = {				
					OR = {
						NOT = { has_government_attribute = th_decreased_danmaku_duel_refusal }
						NOT = { has_country_flag = th_was_already_danmaku_challenged_by_@ROOT }
						had_country_flag = {
							flag = th_was_already_danmaku_challenged_by_@ROOT
							days = 3650
						}
					}
				}
			}
		}
		if = {
			limit = {
				has_country_modifier = th_was_in_danmaku_duel
			}
			custom_trigger_tooltip = {
				tooltip = TH_DANMAKU_HAS_3_YEARS_COOLDOWN_TT
				NOT = { has_country_modifier = th_was_in_danmaku_duel }
			}
		}
		if = {
			limit = {
				alliance_with = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_IS_OUR_ALLY
				NOT = { alliance_with = FROM }
			}
		}
		if = {
			limit = {
				war_with = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_AT_WAR_WITH_FROM
				NOT = { war_with = ROOT }
			}
		}
		if = {
			limit = {
				has_regency = yes
			}
			custom_trigger_tooltip = {
				tooltip = TH_HAS_REGENCY
				has_regency = no
			}
		}
		if = {
			limit = {
				FROM = { has_regency = yes }
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_HAS_REGENCY
				FROM = { has_regency = no }
			}
		}
	}

	on_accept = {
		add_mil_power = -25
		th_getruler_fighting_skill = {
			skill_boni = 0
			type = normal
		}
		th_danmaku_set_compare_variable = {
			type = normal
		}
		FROM = {
			clr_country_flag = th_was_already_danmaku_challenged_by_@ROOT
			clr_country_flag = th_danmaku_duel_lost_against_@FROM
			country_event = {
				id = th_diplo_events.23
				days = 1
			}
			if = {
				limit = { NOT = { has_country_flag = th_has_danmaku_decline_counter } }
				set_country_flag = th_has_danmaku_decline_counter
				set_variable = {
					which = th_declined_danmaku_duel_counter
					value = 0
				}
			}
		}
	}

	ai_will_do = {
		always = no
	}
}
