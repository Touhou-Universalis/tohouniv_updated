  		religion = th_scientistic
		NOT = { has_country_modifier = th_chr_art_of_faith }

		NOT = { has_country_modifier = th_chr_art_of_faith }
  	}

  	}

				is_overseas = no

			}

			factor = 30

				factor = 0

		}

			tradition = 50
		}
		ai_chance = {
			factor = 30
			modifier = {
				factor = 0
				is_at_war = no
			}
		}
	}

	option = {	#The Art of Science
		name = th_scientistic_flavor.30.c
		add_years_of_income = -0.5
		change_innovativeness = 3
		add_country_modifier = {

			factor = 30
		}
	}
