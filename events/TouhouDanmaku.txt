country_event = {
	id = th_danmaku_event.3
	title = "th_danmaku_event.3.t"
	desc = "th_danmaku_event.3.desc"
	picture = SCARLETT_eventPicture
	
	is_triggered_only = yes


	immediate = {
		hidden_effect = {
			set_country_flag = th_just_finished_incident_danmaku_fight
		}
	}
	
	option = {
		name = "th_danmaku_event.3.a"
		th_incident_change_points = {
			value = 2
		}
		th_danmaku_won_a_danmaku_fight = yes
		hidden_effect = {
			set_country_flag = th_incident_last_fight_won
			clr_country_flag = th_incident_last_fight_lost
			clr_country_flag = th_incident_last_fight_terribly_lost
			clr_country_flag = th_just_finished_incident_danmaku_fight
		}
		add_prestige = 10
		add_mil_power = 25
		if = {
			limit = {
        			variable_arithmetic_trigger = {
					export_to_variable = {
						which = challenger_power
                		value = modifier:danmaku_power_constant
					}
					export_to_variable = {
                		which = challenged_power
                		value = modifier:danmaku_power_constant
						who = FROM
					}
					check_variable = {
						which = challenger_power
                		which = challenged_power
					}
				}
			}
			set_variable = {
				which = tempPower
				value = 10
			}
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = roll_power_gain
							value = 0
						}
					}
				}
				set_variable = {
					which = roll_power_gain
					value = 0
				}
			}
			#Power gain for the challenger
			change_variable = {
				which = roll_power_gain
				value = 10    #Whatever you need here
			}
			th_export_root_variable_into_foreign = {
				who = FROM
				root_variable = roll_power_gain
				foreign_variable = tempPower
			}
		}
		else = {
    		from = {
				change_variable = {
					which = roll_power_gain
					value = 10    #Whatever you need here
				}
				th_export_root_variable_into_foreign = {
					who = FROM
					root_variable = roll_power_gain
					foreign_variable = tempPower
				}
			}
		}
	}
}
