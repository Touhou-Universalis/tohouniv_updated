namespace = th_education_events

#Setting up the frequency variable
country_event = {
	id = th_education_events.1
	title = th_education_events.1.t
	desc = th_education_events.1.desc
	picture = MAGICALLIBRARY_eventPicture
	
	trigger = {
        OR = {                
            has_country_flag = th_university_education_enabled
            has_country_flag = th_fairy_university_education_enabled
        }
		NOT = { has_country_flag = th_education_setup }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	hidden = yes
	
	immediate = {
		hidden_effect = {
			set_variable = {
				which = th_education_accelerator_variable
				value = 0
			}
			set_country_flag = th_education_setup
        }
	}
	
	option = {
		name = th_education_events.1.a
		
		ai_chance = { 
			factor = 100
		}
	}
}

#General Education
country_event = {
	id = th_education_events.2
	title = th_education_events.2.t
    desc = {
       trigger = { NOT = { has_country_flag = th_fairy_university_education_enabled } }
       desc = th_education_events.2.desc
    }
    desc = {
       trigger = { has_country_flag = th_fairy_university_education_enabled }
       desc = th_education_events.2.fairy.desc
    }
	desc = th_education_events.2.desc
	picture = MAGICALLIBRARY_eventPicture
	
	trigger = {
        OR = {
            has_country_flag = th_university_education_enabled
            has_country_flag = th_fairy_university_education_enabled
        }
        is_lesser_in_union = no
        has_country_flag = th_education_setup
	}
	
	mean_time_to_happen = {
		months = 1200
		
		modifier = {
			factor = 0.9
			AND = {
				check_variable = {
					which = th_education_accelerator_variable
					value = 1
				}
				NOT = {
					check_variable = {
						which = th_education_accelerator_variable
						value = 2
					}
				}
			}
		}
		
		modifier = {
			factor = 0.75
			AND = {
				check_variable = {
					which = th_education_accelerator_variable
					value = 2
				}
				NOT = {
					check_variable = {
						which = th_education_accelerator_variable
						value = 3
					}
				}
			}
		}
		
		modifier = {
			factor = 0.6
			AND = {
				check_variable = {
					which = th_education_accelerator_variable
					value = 3
				}
				NOT = {
					check_variable = {
						which = th_education_accelerator_variable
						value = 4
					}
				}
			}
		}
		
		modifier = {
			factor = 0.45
			AND = {
				check_variable = {
					which = th_education_accelerator_variable
					value = 4
				}
				NOT = {
					check_variable = {
						which = th_education_accelerator_variable
						value = 5
					}
				}
			}
		}
		
		modifier = {
			factor = 0.3
			AND = {
				check_variable = {
					which = th_education_accelerator_variable
					value = 5
				}
				NOT = {
					check_variable = {
						which = th_education_accelerator_variable
						value = 6
					}
				}
			}
		}
		
		modifier = {
			factor = 0.15
			AND = {
				check_variable = {
					which = th_education_accelerator_variable
					value = 6
				}
			}
		}
	}
	
	option = {	#Focus on admin
        name = th_education_events.2.a
        if = {
            limit = {
                has_country_flag = th_fairy_university_education_enabled
            }
            th_increase_ruler_adm_stat = {
                adm = 1
            }
        }
        else = {
            if = {
                limit = {
                    NOT = { adm = 6 }
                }
                change_adm = 1
            }
            else = {
                add_adm_power = 75
            }
        }
		add_years_of_income = -0.15
	}
	
	option = {	#Focus on dip
		name = th_education_events.2.b
        if = {
            limit = {
                has_country_flag = th_fairy_university_education_enabled
            }
            th_increase_ruler_dip_stat = {
                dip = 1
            }
        }
        else = {
            if = {
                limit = {
                    NOT = { dip = 6 }
                }
                change_dip = 1
            }
            else = {
                add_dip_power = 75
            }
        }
		add_prestige = -10
		add_legitimacy = -10
	}
	
	option = {	#Focus on mil
		name = th_education_events.2.c
        if = {
            limit = {
                has_country_flag = th_fairy_university_education_enabled
            }
            th_increase_ruler_mil_stat = {
                mil = 1
            }
        }
        else = {
            if = {
                limit = {
                    NOT = { mil = 6 }
                }
                change_mil = 1
            }
            else = {
                add_mil_power = 75
            }
        }
		add_yearly_manpower = -0.1 
	}
	
	option = {	#Let her rest
		name = th_education_events.2.d
		add_adm_power = 25
		add_dip_power = 25
		add_mil_power = 25
	}
}
