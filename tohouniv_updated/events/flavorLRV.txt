country_event = {
	id = flavor_lrv.6
	title = flavor_lrv.6.t
	desc = flavor_lrv.6.desc
	picture = ETERNITYGODDESS_eventPicture

	trigger = {
		has_reform = Gensokyan_lrv_divine_government
		OR = {
			NOT = { has_country_flag = lrv_had_larva_deity_event }
			AND = {
				had_country_flag = {
					flag = lrv_had_larva_deity_event
					days = 1460
				}
				has_personal_deity = Larva
			}
			AND = {
				had_country_flag = {
					flag = lrv_had_larva_deity_event
					days = 3650
				}
				NOT = { has_personal_deity = Larva }
			}
		}
		NOT = { has_country_flag = th_deity_penalty_active }
		NOT = { has_country_flag = th_having_a_deity_event }
		is_at_war = yes
		NOT = { has_country_modifier = th_deity_eternal_warriors }
	}

	immediate = {
		clr_country_flag = lrv_had_larva_deity_event
		set_country_flag = th_having_a_deity_event
	}

	after = {
		set_country_flag = lrv_had_larva_deity_event
		clr_country_flag = th_having_a_deity_event
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	option = {
		name = flavor_lrv.6.a
		add_country_modifier = {
			name = th_deity_eternal_warriors
			duration = 1825
		}
		add_war_exhaustion = 1

		name = flavor_lrv.6.b

}
