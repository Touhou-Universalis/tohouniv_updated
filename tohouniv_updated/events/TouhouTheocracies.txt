country_event = {	#Incident - Non-th_hakureishinto declares war upon Gensokyo
	id = touhou_theocracy.2
	title = touhou_theocracy.2.t
	desc = touhou_theocracy.2.desc
	picture = SHRINE_eventPicture
	
	mean_time_to_happen = {
		months = 1
	}
	
	trigger = {
		NOT = { capital_scope = { continent = gensokyo_continent } }
		NOT = { religion_group = Touhou }
		NOT = { has_country_flag = TH_foreign_invader }
		any_country = {
			capital_scope = { continent = gensokyo_continent }
			NOT = { has_country_flag = TH_asked_for_help_against_@ROOT }
			is_in_war = {
				attacker_leader = ROOT
				defender_leader = THIS
			}
		}
		any_country = {
			capital_scope = { continent = gensokyo_continent }
			has_reform = Gensokyan_shrine_theocracy
		}
	}
	
	option = {		#Defender will call now the shrine maiden
		name = touhou_theocracy.2.a
		custom_tooltip = touhou_theocracy.2.a.tt
		hidden_effect = {
			set_country_flag = TH_foreign_invader
			every_country = {
				limit = {
					capital_scope = { continent = gensokyo_continent }
					NOT = { has_country_flag = TH_asked_for_help_against_@ROOT }
					is_in_war = {
						attacker_leader = ROOT
						defender_leader = THIS
					}
					OR = {
						is_subject = no
						is_subject_of_type = tributary_state
					}
				}
				country_event = { id = touhou_theocracy.3 }
				set_country_flag = TH_asked_for_help_against_@ROOT
			}
		}
		ai_chance = {
			factor = 100
		}
	}
}
