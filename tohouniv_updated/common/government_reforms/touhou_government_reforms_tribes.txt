TH_fairies_become_absolute_monarchy_reform = {
	icon = "queen"
	allow_normal_conversion = yes
	potential = {
		primary_culture = Fairy
	}
	effect = {
		set_country_flag = populists_in_government
		lose_reforms = 2
		change_government = monarchy
		if = {
			limit = {
				th_is_or_was_tag = {
					tag = CIN
				}
			}
			add_government_reform = Gensokyan_cirno_kingdom_government
		}
		else = {
			add_government_reform = Gensokyan_monarchy_fairy_government
		}
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_global_flag = th_royal_fairy_challenge_happened_once }
				}
				set_global_flag = th_royal_fairy_challenge_is_about_to_happen
				if = {
					limit = {
						th_is_or_was_tag = {
							tag = CIN
						}
					}
					country_event = {
						id = flavor_cin.3
						days = 365
					}
				}
				else = {
					country_event = {
						id = rfc_events.15
						days = 365
					}
				}
			}
		}
	}
	ai = {
		factor = 500
	}
}

	}
}
