###PLEDGED TO RELIGION###
th_religious_pledged = {
	custom_trigger_tooltip = {
		tooltip = th_pledged_to_religion_tt
		OR = {
			has_country_flag = th_pledged_to_religion_to_@$tag$
			AND = {
				is_subject_of = $tag$
				has_country_modifier = th_subjugated_in_penance
			}
		}
	}
}
