th_is_not_major_main_character = {	#Use "th_is_not_major_main_character = no" if the priority is irelevant
	NOT = { tag = HKR }	#Reimu
	NOT = { tag = FOM }	#Marisa
	
	NOT = { was_tag = HKR }
	NOT = { was_tag = FOM }
}

th_is_major_main_character = {		#Gives HKR a higher priority than FOM
	OR = {
		tag = HKR
		AND = {
			capital_scope = {
				superregion = gensokyo_superregion
			}
			NOT = { exists = HKR }
			was_tag = HKR
		}

		AND = {
			OR = {
				NOT = { exists = HKR }
				HKR = {
					is_subject_other_than_tributary_trigger = yes
				}
			}
			all_country = {
				limit = {
					capital_scope = {
						superregion = gensokyo_superregion
					}
				}
				NOT = { was_tag = HKR }
			}
			OR = {
				tag = FOM
				AND = {
					capital_scope = {
						superregion = gensokyo_superregion
					}
					NOT = { exists = FOM }
					was_tag = FOM
				}
			}
		}
	}
}

th_is_not_incident_antagonist = {
	if = {	#Is not Remilia
		limit = {
			has_global_flag = th_current_incident_scarlet_mists
		}
		NOT = { tag = SCR }
		NOT = { tag = SCD }

		NOT = { was_tag = SCR }
		NOT = { was_tag = SCD }
	}
	if = {	#Is not Yuyuko or Yukari
		limit = {
			has_global_flag = th_current_incident_cherry_blossom
		}
		NOT = { tag = NMG }
		NOT = { tag = MHG }
		NOT = { tag = NET }

		NOT = { was_tag = NMG }
		NOT = { was_tag = MHG }
		NOT = { was_tag = NET }
	}
	if = {	#Is not Keine, Kaguya or Mokou
		limit = {
			has_global_flag = th_current_incident_eternal_night
		}
		NOT = { tag = EIT }
		NOT = { tag = FUJ }
		NOT = { tag = HMC }

		NOT = { was_tag = EIT }
		NOT = { was_tag = FUJ }
		NOT = { was_tag = HMC }
	}
	if = {	#Is not Aki, Hina, Nitori, Aya or Sanae
		limit = {
			has_global_flag = th_current_incident_mountain_of_faith
		}
		NOT = { tag = AKI }
		NOT = { tag = YOF }
		NOT = { tag = KPP }
		NOT = { tag = TGU }
		NOT = { tag = MRY }

		NOT = { was_tag = AKI }
		NOT = { was_tag = YOF }
		NOT = { was_tag = KPP }
		NOT = { was_tag = TGU }
		NOT = { was_tag = MRY }
	}
}
