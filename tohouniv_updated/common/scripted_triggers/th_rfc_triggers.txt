#12 because a year has 4 cycles (3 month = 1 cylce) and RFC should be 3 years long -> 12 cycles.

th_rfc_has_reached_max_cycles = {
	check_variable = {
		which = th_rfc_current_cycle_of_challenge
		value = 12
	}
}

th_rfc_has_not_reached_max_cyles = {
	NOT = {
		check_variable = {
			which = th_rfc_current_cycle_of_challenge
			value = 12
		}
	}
}
