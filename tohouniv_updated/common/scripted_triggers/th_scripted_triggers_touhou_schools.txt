#Scholars from where?
#Country Scope
th_scholar_from_our_school_trigger = {
	has_country_modifier = $school$_scholar_modifier
	ROOT = {
		religious_school = {
			group = Touhou
			school = $school$_school
		}
	}
}
th_our_scholar_matches_their_school_trigger = {
	PREV = { has_country_modifier = $school$_scholar_modifier }
	religious_school = {
		group = Touhou
		school = $school$_school
	}
}

th_can_invite_scholar_diplomatically = {
	if = {
		limit = {
			exists = $school_country$
		}
		$school_country$ = {
			has_opinion = {
				who = ROOT
				value = 150
			}
			OR = {
				alliance_with = ROOT
				is_subject_of = ROOT
				overlord_of = ROOT
			}
		}
	}
	else = {
		$school_province$ = {
			owner = {
				has_opinion = {
					who = ROOT
					value = 150
				}
				OR = {
					alliance_with = ROOT
					is_subject_of = ROOT
					overlord_of = ROOT
				}
			}
		}
	}
}
