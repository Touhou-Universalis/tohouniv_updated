#MRY request another country to convert
th_request_conversion = {
	category = th_diplo_action
	alert_index = 30
	alert_tooltip = th_request_conversion_tooltip
	require_acceptance = yes

	is_visible = {
		is_free_or_tributary_trigger = yes
		OR = {
			AND = {
				th_is_or_was_tag = { tag = MRY }
				has_reform = Gensokyan_shrine_theocracy
				religion = th_moriyashinto #Just to make it easier
			}
			th_has_copied_from_target = {
				target = MRY
			}
		}
		FROM = {
			NOT = { religion = ROOT }
			any_owned_province = { religion = ROOT }
			NOT = { tag = PAP }
		}
	}

	is_allowed = {
		if = {
			limit = {
				th_has_incidents_enabled = yes
			}
			custom_trigger_tooltip = {
				tooltip = th_mry_conversion_enabled_tt
				has_global_flag = th_mry_conversion_enabled
			}
		}
		else = {
			has_reform = TH_feudal_shrine_reform
		}
		FROM = {
			dominant_religion = th_moriyashinto
			is_free_or_tributary_trigger = yes
			NOT = { tag = PAP }
		}
		if = {
			limit = {
				FROM = {
					OR = {
						has_country_flag = th_immunity_to_diplomatic_actions
						has_government_attribute = enables_yama_authority
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		add_prestige = 10
		FROM = {
			change_religion = ROOT
			add_opinion = {
    			who = ROOT
    			modifier = th_great_guider
			}
		}
		add_opinion = {
    		who = FROM
    		modifier = th_new_follower
		}
		mry_add_scaled_country_conversion_faith_power = {
			enforcer = ROOT
			converted = FROM
			value = 1
			development_factor = 1
		}
	}

	on_decline = {
		FROM = {
			add_opinion = {
    			who = ROOT
    			modifier = th_instigator
			}
		}
		add_opinion = {
    		who = FROM
    		modifier = th_refused_conversion
		}
	}

    ai_acceptance = {
        add_entry = {
            name = TH_DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
		add_entry = {
			name = TH_DIPLO_TRUST_WITH
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = TH_DIPLO_TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			#1 dev = -0.5
			multiply_variable = {
				which = ai_value
				value = -0.5
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_hostile
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_outraged
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000
			}
		}
		add_entry = {
			name = DIPLO_IS_RIVAL
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_rivalry
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000

		}
		add_entry = {

				variable_name = ai_value

						who = ROOT

				variable_name = ai_value

			limit = { }

		}

			limit = {
				OR = { 
					historical_rival_with = FROM
					FROM = { historical_rival_with = ROOT }

#MRY incite Moriya zealots
th_incite_moriya_zealots = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		is_free_or_tributary_trigger = yes
		OR = {
			AND = {
				th_is_or_was_tag = { tag = MRY }
				has_reform = Gensokyan_shrine_theocracy
				religion = th_moriyashinto #Just to make it easier
			}
			th_has_copied_from_target = {
				target = MRY
			}
		}
		FROM = {
			NOT = { religion = ROOT }
			any_owned_province = { religion = ROOT }
		}
	}

	is_allowed = {
		has_spy_network_in = {
			who = FROM
			value = 50
		}
		dip_tech = 9
		custom_trigger_tooltip = {
			tooltip = th_diplomacy_has_already_incited_zealots
			FROM = {				
				OR = {
					NOT = { has_country_flag = th_has_already_incited_zealots }
					had_country_flag = {
						flag = th_has_already_incited_zealots
						days = 1826
					}
				}
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		FROM = {
			set_country_flag = th_has_already_incited_zealots
			every_owned_province = { 
				limit = {
					religion = ROOT
				}
				add_province_modifier = { 
    				name = th_incited_zealots
    				duration = 1825  
				}
			}
			country_event = {
				id = th_diplo_events.12
			}
		}
		add_spy_network_in = {
			who = FROM
			value = -50			
		}
	}

	ai_will_do = {
		always = no
	}
}
