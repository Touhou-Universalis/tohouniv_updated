th_religious_center = {
	potential = {
	}
	
	#viewer = {
	#}
	
	trigger = {
		if = {
			limit = {
				province_id = 4803 # Hakurei Shrine
			}
			religion = th_hakureishinto
		}
		else_if = {
			limit = {
				province_id = 4846 # Moriya Shrine
			}
			religion = th_moriyashinto
		}
		else_if = {
			limit = {
				province_id = 4835 # Myouren Temple
			}
			religion = th_buddhist
		}
		else_if = {
			limit = {
				province_id = 4873 # Divine Spirit Mausoleum
			}
			religion = th_taoist
		}
		else_if = {
			limit = {
				province_id = 5031 # Old Hakurei Shrine
			}
			religion = th_oldhakurei
		}
		else_if = {
			limit = {
				province_id = 5053 # Fallen Temple
			}
			religion = th_shinkism
		}
		else_if = {
			limit = {
				province_id = 5149 # Probability Space Vessel
			}
			religion = th_christian
		}
		else_if = {
			limit = {
				province_id = 5168 # Lunar Capital
			}
			religion = th_lunarian_religion
		}
	}

	local_missionary_strength = -0.05
	
	#Province scope
	on_activation = {
	
	}
	
	on_deactivation = {
	
	}
}
